template<typename T >
template<std::input_or_output_iterator TIt>
std::shared_ptr<void> dl::utils::GenericIterator< T >::Iter< TIt >::userdata
private

A shared_ptr to data that should be deleted when the iterator is deleted.

This is intended to store the underlying data into that the iterator iterates over. As such, this allows returning the begin-end-iterator-pair from a function and keeping them point to valid memory until the last of the iterators is destructed.

auto getData() {
auto data = std::make_shared<std::vector>({ 1, 2, 3, 4 });
return std::make_tuple(Iter(data->begin(), data), Iter(data->end(), data));
}
